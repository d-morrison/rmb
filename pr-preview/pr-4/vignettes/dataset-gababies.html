var_labels <- vapply(dat, function(x) {
  lbl <- attr(x, "label")
  if (is.null(lbl) || length(lbl) == 0 || all(lbl == "")) NA_character_ else paste(as.character(lbl), collapse = "; ")
}, character(1))

3 Example model

Code
num_vars <- names(dat)[vapply(dat, is.numeric, logical(1))]
if (length(num_vars) >= 2) {
  response <- num_vars[1]
  predictor <- num_vars[2]
  df <- stats::na.omit(dat[, c(response, predictor), drop = FALSE])
  if (nrow(df) >= 5) {
    y <- df[[response]]
    if (all(y %in% c(0, 1))) {
      fit <- stats::glm(stats::as.formula(paste(response, predictor, sep = " ~ ")), data = df, family = stats::binomial())
    } else {
      fit <- stats::lm(stats::as.formula(paste(response, predictor, sep = " ~ ")), data = df)
    }
    summary(fit)
  } else {
    cat("Not enough complete observations for a model.\n")
  }
} else {
  cat("No two numeric variables available for a simple model example.\n")
}
